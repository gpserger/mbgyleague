CREATE TABLE `mbgyleague`.`user` ( `UserID` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT , `Name` VARCHAR(200) NOT NULL , `Username` VARCHAR(200) NOT NULL , `Email` VARCHAR(200) NOT NULL , `Password` VARCHAR(200) NOT NULL , `Userlevel` INT(3) UNSIGNED NOT NULL DEFAULT '1' , `reg_date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`UserID`)) ENGINE = InnoDB;



CREATE TABLE `mbgyleague`.`lolplayers` ( `id` INT(11) UNSIGNED NOT NULL , `summoner_name` VARCHAR(20) NOT NULL , `elo` INT(4) UNSIGNED NOT NULL DEFAULT 1200, PRIMARY KEY (`id`)) ENGINE = InnoDB;
CREATE TABLE `mbgyleague`.`lolmatches` ( `MatchID` INT(11) UNSIGNED NOT NULL AUTO_INCREMENT , `player1id` INT(11) UNSIGNED NOT NULL , `player2id` INT(11) UNSIGNED NOT NULL , `winner` INT(11) UNSIGNED NOT NULL , `played_date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`MatchID`)) ENGINE = InnoDB;
